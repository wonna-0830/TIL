```java
import java.util.*; 
class Solution { 
    public int[] solution(int[] arr, int divisor) { 
        List<Integer> answer = new ArrayList<>(); 
        for(int i: arr){ 
            if(i % divisor == 0){
                answer.add(i); 
            } 
        } 
        if(answer.size() == 0){
            answer.add(-1); 
        } 
        answer.sort(Comparator.naturalOrder()); 
        return answer.stream().mapToInt(i->i).toArray();
    }    
}
```

```java
import java.util.*;
class Solution {
    public int[] solution(int[] arr, int divisor) {
        int num = 0;
        for(int i : arr){
            if(i % divisor == 0){
                num += 1;
            }
        }
        if(num == 0){
            return new int[]{-1};
        } 
        
        int[] answer = new int[num];
        int idx = 0;
        
        for(int i : arr){
            if(i % divisor == 0){
                answer[idx++] = i;
            }
        }
        
        Arrays.sort(answer);
        return answer;
    }
}
```

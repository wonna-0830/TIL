class Solution {
    public int solution(String t, String p) {
        int n = t.length();
        int m = p.length();
        int answer = 0;

        for (int i = 0; i <= n - m; i++) {
            String sub = t.substring(i, i + m); 
            if (sub.compareTo(p) <= 0) {        
                answer++;
            }
        }
        return answer;
    }
}

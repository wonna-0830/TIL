import java.util.*;
class Solution {
    public boolean solution(int x) {
        boolean answer = true;
        String[] arr = String.valueOf(x).split("");
        int num = 0;
        for(int i = 0; i < arr.length; i++){
            num += Integer.parseInt(arr[i]);
        }
        
        if(x % num == 0){
            answer = true;
        } else {
            answer = false;
        }
        return answer;
    }
}
